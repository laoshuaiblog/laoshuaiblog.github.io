4.将数组a中的n个整数按逆序存放。即：a[0]与a[n-1]交换，a[1]与a[n-2]交换......。交换要调用函数 void(int *a,int n)。

#include<stdio.h>
#define N 10
void sort(int *a,int n)
{
    int i,j,temp;
    for(i=0,j=n-1; i<n/2,j>=n/2; i++,j--)
    {
        temp=a[j];
        a[j]=a[i];
        a[i]=temp;

    }


}
int main()
{
    int a[N],i,n=10;

    printf("输入10个数字:");
    for(i=0; i<N; i++)
    {
        scanf("%d",&a[i]);
    }

    sort(&a,n);
    for(i=0; i<N; i++)
    {
        printf("%d ",a[i]);

    }
    return 0;
}
