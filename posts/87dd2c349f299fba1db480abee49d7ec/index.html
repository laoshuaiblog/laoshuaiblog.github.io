使用原生nodejs搭建一个简易的web服务器demo

简易demo

var http = require('http');
var url = require("url");
const app= http.createServer(function (request, response) {
    var urlObj = url.parse(request.url,true);
    console.log(request.url);
    // 内容类型: text/plain。并用charset=UTF-8解决输出中文乱码
    response.writeHead(200, {'Content-Type': 'text/plain; charset=UTF-8'});
 
    // 下句是发送响应数据
    response.end('Hello World! 这是简单的web服务器测试。\n');
}).listen(8888);
// 终端打印如下信息
console.log('Server running at http://127.0.0.1:8888/');

使用node作为中转服务器

访问http://127.0.0.1:8888/list就相当于请求了https://api.juejin.cn/tag_api/v1/query_category_briefs的接口

var http = require("http");
const https = require("https");
var url = require("url");
const app = http.createServer();

app.on("request", (req, res) => {
  res.writeHead(200, { "Content-Type": "text/plain; charset=UTF-8" });
  let path = url.parse(req.url, true).pathname;

  console.log(123, req.url, req.method, path);
  switch (path) {
    case "/":
      res.end("hello world --index--");

      break;
    case "/list":
      https
        .get(
          "https://api.juejin.cn/tag_api/v1/query_category_briefs",
          (result) => {
            result.on("data", (data) => {
              // console.log(data);//Buffer数据
              process.stdout.write(data); //更底层，输出结果等同于console.log(data.toString());
              // console.log底层就是基于 process.stdout.write 实现
              res.end(data);
            });
          }
        )
        .on("error", (err) => {
          console.log("Error: ", err.message);
        });
      break;
    default:
      //   res.writeHead(404);
      res.write("你访问的页面是" + path);
      res.end();
  }
});

app.listen(8888);
// 终端打印如下信息
console.log("Server running at http://127.0.0.1:8888/");
