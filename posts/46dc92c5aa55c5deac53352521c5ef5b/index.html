#pragma once
#include <iostream>
using namespace std;
#define MAX 100//最大人数
//设计联系人的结构体
struct Person
{
	string m_Name;//姓名
	int m_Sex;//性别:1男 2女
	int m_Age;//年龄
	string m_Phone;//电话
	string m_Addr;//家庭住址
};
//设计通讯录的结构体
struct Addressbooks
{
	Person personArray[MAX];//通讯录中保存的联系人数组
	int m_Size;//通讯录中当前记录的联系人个数
};

void showMenu();
void addPerson(Addressbooks* abs);
void showPerson(Addressbooks* abs);
int isExist(Addressbooks* abs, string name);
void deletePerson(Addressbooks* abs);
void findPerson(Addressbooks* abs);
void modifyPerson(Addressbooks* abs);
void cleanPerson(Addressbooks* abs);

 3.添加联系人.cpp

#include <iostream>
#include <string>
#include "通讯录管理系统.h"
using namespace std;

void addPerson(Addressbooks* abs)
{
	//通讯录是否已满,如果满了就不再添加
	if (abs->m_Size == MAX)
	{
		cout << "通讯录已满,无法添加!" << endl;
		return;
	}
	else
	{
		//添加具体联系人
		//姓名
		string name;
		cout << "请输入姓名:" << endl;
		cin >> name;
		abs->personArray[abs->m_Size].m_Name = name;

		//性别
		cout << "请输入性别:" << endl;
		cout << "1 --- 男" << endl;
		cout << "2 --- 女" << endl;
		int sex = 0;
		while (true)
		{/*如果输入的是1或者2可以退出循环,如果输入其它,则重新输入*/
			cin >> sex;
			if (sex == 1 || sex == 2)
			{
				abs->personArray[abs->m_Size].m_Sex = sex;
				break;
			}
			cout << "输入有误,请重新输入:" << endl;
		}

		//年龄
		cout << "请输入年龄:" << endl;
		int age = 0;
		while (true)
		{/*年龄必须是整数,输入负数则重新输入*/
			cin >> age;
			if (age>0)
			{
				abs->personArray[abs->m_Size].m_Age = age;
				break;
			}
			cout << "输入有误,请重新输入:" << endl;
		}

		//电话
		cout << "请输入联系电话:" << endl;
		string phone;
		cin >> phone;
		abs->personArray[abs->m_Size].m_Phone = phone;

		//住址
		cout << "请输入住址:" << endl;
		string address;
		cin >> address;
		abs->personArray[abs->m_Size].m_Addr = address;

		//更新通讯录中的人数
		abs->m_Size++;
		cout << "添加成功!" << endl;
		system("pause");//请按任意键继续
		system("cls");//清屏
	}
}

 5.删除联系人.cpp

#include <iostream>
#include <string>
#include "通讯录管理系统.h"
using namespace std;

void deletePerson(Addressbooks* abs)
{
	cout << "请输入您要删除的联系人" << endl;
	string name;
	cin >> name;
	int ret = isExist(abs, name);
	if (ret == -1)
	{
		cout << "查无此人" << endl;
	}
	else
	{/*找到此人进行删除*/
		for (int i = ret ; i < abs->m_Size-1; i++)
		{
			abs->personArray[i] = abs->personArray[i + 1];//数据前移
		}
		abs->m_Size--;//更新通讯录
		cout << "删除成功" << endl;
	}
	system("pause");
	system("cls");
}

 6.检测联系人是否存在.cpp

#include <iostream>
#include <string>
#include "通讯录管理系统.h"
using namespace std;
/*检测联系人是否存在,如果存在返回联系人的位置,不存在返回-1*/
int isExist(Addressbooks* abs,string name)
{/*参数一:通讯录指针
 参数二:要查找的联系人名字*/
	for (int i = 0; i < abs->m_Size; i++)
	{//遍历查找
		if (abs->personArray[i].m_Name == name)
		{
			return i;//找到返回下标
		}
	}
	return -1;//没找到返回-1
}

 10.主函数

#include <iostream>
#include <string>
using namespace std;
#include "通讯录管理系统.h"

int main()
{
	Addressbooks abs;//创建通讯录结构体变量
	abs.m_Size = 0;
	int select = 0;
	while (true)
	{	
		showMenu();
		cin >> select;
		switch (select)
		{
		case 1://1.添加联系人
			addPerson(&abs);//利用地址可以修改实参
			break;
		case 2://2.显示联系人
			showPerson(&abs);
			break;
		case 3://3.删除联系人
			deletePerson(&abs);
			break;
		case 4://4.查找联系人
			findPerson(&abs);
			break;
		case 5://5.修改联系人
			modifyPerson(&abs);
			break;
		case 6://6.清空联系人
			cleanPerson(&abs);
			break;
		case 0://0.退出通讯录
			cout << "欢迎下次使用" << endl;
			system("pause");
			return 0;
			break;
		default:
			break;
		}
	}

	system("pause");
	return 0;
}
