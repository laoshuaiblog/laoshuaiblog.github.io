如何写漂亮的返回值为 boolean 类型的方法

public boolean isLow() { 
    if ((myGasoline + myBattery) < 10) { 
        return true; 
    } else { 
        return false; 
    } 
} 

// 更漂亮的：
public boolean isLow() { 
    if ((myGasoline + myBattery) < 10) { 
        return true; 
    }
    return false;
} 

// 更漂亮的：

public boolean isLow() { 
    // Compute (true or false) if we are low, 
    // and return that boolean value directly. 
    return ((myGasoline + myBattery) < 10);
}
