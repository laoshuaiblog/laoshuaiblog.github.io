阿姆斯特朗数也就是俗称的水仙花数，是指一个三位数，其各位数字的立方和等于该数本身。例如：153=1^3+5^3+3^3，所以 153 就是一个水仙花数。求出所有的水仙花数。

#include <stdio.h>
#include <math.h>

int main()
{
    int i,x,y,z;
    for(i=100;i<1000;i++)
    {
        x=i%100%10;
        y=i/10%10;
        z=i/100;
        if(pow(x,3)+pow(y,3)+pow(z,3)==i)
            printf("%d ",i);
    }
    return 0;
}
