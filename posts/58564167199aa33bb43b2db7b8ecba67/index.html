删除树形结构中的某个属性

arr为需处理的数组 key为要删除的属性名

 const handler = (arr, key) => {
    if (arr) {
      arr.forEach((item) => {
      	if(item[key]){
      		delete item[key]
      	}
        if (item.children && item.children!==[]) handler(item.children, key)
      })
      return arr
    }
  }
