本关任务：下面的程序是从键盘输入三个整数按从小到大的顺序输出。

#include<stdio.h>
	int main()
	{  
	  int a,b,c,t;
	  printf("请输入三个整数:");
      scanf("%d%d%d",&a,&b,&c);
	  if(a>b)
	  {
		  t=a;
		  a=b;
		  b=t;
	  }
	  if(a>c)
	  {
		  t=a;
		  a=c;
		  c=t;
	  }
	  if(b>c)
	  {
		  t=b;
		  b=c;
		  c=t;
	  }
	  printf("从小到大排列为:%d,%d,%d",a,b,c);
       return 0;
	}
