Rust引用借用 vs 原始指针

let v: i32 = 10;
let v1 = &v;
let a: *const i32 = &v;
println!("a={:p}", a);
let b = a as *mut i32;  //无法从&v直接转换成*mut i32，需要从*const i32过度
println!("b={:p}", b);
unsafe {
    *b+=1;
}
println!("v={}   v1={}", v, v1);
