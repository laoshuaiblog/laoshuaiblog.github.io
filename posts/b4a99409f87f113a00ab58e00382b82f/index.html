C语言：写一个函数求字符串长度

#include<stdio.h>
#include<assert.h>
size_t my_strlen(const char* str) {
	assert(str);
	size_t count = 0;
	while (*str!='\0') {
		str++;
		count++;
	}
	return count;
}
int main() {
	char arr[] = "abcdef";
	size_t n = my_strlen(arr);
	printf("%u\n", n);
}
