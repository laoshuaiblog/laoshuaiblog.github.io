九进制以内的乘法口诀表

 本代码只适用于九进制以内，更高进制在下篇

#include"bits/stdc++.h"

using namespace std;

int p,m,n,c,sum;
int main()
{
	cin >> p;
	for(int i=1;i<=9;i++)
	{
		for(int j=1;j<=i;j++)
		{
			c=0;
			sum=0;
			n=i*j;
			while(p)//设置一个无限循环
			{
				
				m=n%p;
				sum=m*pow(10,c)+sum;//进制转换，再开辟一个函数会更直观
				if(n<p) break;
				n=n/p;//短除法
				c++;
				
			}
			if(j<i)
			printf("%d*%d=%d ",i,j,sum);//注意形式
			else
			printf("%d*%d=%d\n",i,j,sum);
		}
	}
	return 0;
}
