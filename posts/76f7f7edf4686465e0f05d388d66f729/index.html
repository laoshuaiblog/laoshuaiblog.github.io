#include<bits/stdc++.h>
#define int long long
#define endl '\n'
#define INF 0x3f3f3f3f
using namespace std;
 
const int N = 100010;
int n, x,num,ans;
signed main(){
	std::ios::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);
	priority_queue<int, vector<int>,greater<int>> pq;
	cin>>n;
	while(n--){
		cin>>num;
		pq.push(num);
	}
	while(pq.size()>1){
		int a = pq.top();
		pq.pop();
		int b = pq.top();
		pq.pop();
		pq.push(a+b);
		ans =ans+ a+b;
		
	}
	cout<<ans;
}
