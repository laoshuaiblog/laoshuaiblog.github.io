C语言：顺序查找

#include<stdio.h>
main()
{
int i,x,j=0;
int a[10]={12,34,64,73,23,56,43,54,32,56};
printf("请输入要查找的数：\n");
scanf("%d",&x);
for(i=0;i<10;i++)
{
if(a[i]==x)
printf("%d ",i);
else 
j++;
}
if(j>=10)
printf("没有找到");
}
